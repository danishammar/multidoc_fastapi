
# Multipdf_chatbot
## Stack:
1. app framework = fastapi
2. llm framework = langchain
3. embedding = HuggingFace
4. vector db = chroma
5. pdfloader = PyPDF Loader
6. llm = mGroq(llama-3.1-70b-versatile)
